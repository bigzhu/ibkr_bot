        if price >= base_price:

        # 获取最低持仓成本作为计算基础
        cost_price = _get_lowest_unmatched_price(unmatched_orders)

            cost_price = Decimal(str(demark_klines[0]["high"]))

        # 根据最低持仓成本计算可购买数量: base_quantity * cost_price = user_balance

## 平均分配, 尽量使用所有资金

```python
    # BUY 订单金额计算: 动态模式 - 根据价格下跌幅度动态调整投入资金
    base_price = _get_highest_unmatched_price(unmatched_orders)
    if base_price is None:
        base_price = Decimal(str(demark_klines[0]["high"]))

    last_price = _get_lowest_unmatched_price(unmatched_orders)
    if last_price is None:
        last_price = Decimal(str(demark_klines[0]["high"]))

    # 价格校验: 本次买入价格不应高于上次最低买入价格
    if price > last_price:
        raise ValueError(
            f"价格反弹,本次买入价格({price})高于上次最低买入价格({last_price}),不应继续买入"
        )

    top_price = base_price
    floor_price = base_price / Decimal(10)  # 下限: base_price/10

    locked_balance = get_locked_balance(
        symbol, timeframe
    )  # 查询被套住的余额,用于计算可用资金
    all_balance = user_balance + locked_balance

    # 根据下跌幅度均匀分布投入资金
    price_drop = last_price - price
    price_range = top_price - floor_price
    drop_ratio = price_drop / price_range
    entry_amount = all_balance * drop_ratio
```
